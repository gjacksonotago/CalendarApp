302 /*
303  * Draws the calendar dates to the canvas from inside the calendar app
304  * Takes number of days in the month and the day to start on
305  * 
306  * @param {int} daysInmonth
307  * @param {int} startDay
308  */
309 function drawCalendar(daysInMonth, startDay) {
310     //Populates the 'screen' with clickable calendar date icons
311     var days = 0;
312     var beginDays = false;
313     var gapSize = 40;//distance between individual date squares
314     var daysOfWeek = ['S', 'M', 'T', 'W', 'T', 'F', 'S'];
315     dayNo[0] = 0;
316 
317     var j, i;
318     for (j = 0; j < 7; j += 1) {
319         for (i = 0; i < 7; i += 1) {
320             if (j === 0) {
321                 //Smaller squares for the days of the week at the beginning
322                 drawRect((gapSize * i) + 20, (gapSize * j) + 40, 30, 15, "#000000");
323 
324             } else {
325                 //Larger boxes for the actual days - because otherwise a full month
326                 //doesn't fit on the "screen"
327                 //drawRect((gapSize * i) + 20, (gapSize * j) + 20, 30, 30, "#FFFFFF");
328             }
329             //Writes the days of the week text.
330             if (j === 0) {
331                 writeSomething(daysOfWeek[i], (gapSize * i) + 25, (gapSize * j) + 50, 8);
332             }
333             if (i === startDay && j === 1) {
334                 beginDays = true;
335             }
336             if (j > 0 && days < daysInMonth && beginDays) {
337                 var jcoord = (gapSize * j) + 20;
338                 var icoord = (gapSize * i) + 20;
339 
340                 days += 1;
341                 dayNo[days] = days;
342                 var key = icoord + "" + jcoord;
343                 reminderCoords[key] = days;
344 
345                 //Function wrapped to prevent evaluation on passing as function.
346                 //Please don't make this a "new function" ever.
347                 var addRem = function (x, y) {
348                     var r = x + "" + y;
349                     addReminder(reminderCoords[r]);
350                 };
351                 //Enlarges the date when mouse pressed down over it
352                 var enlarge = function (x, y, day) {
353                     drawRect(x - 15, y - 15, 60, 60, "black");
354                     writeSomething(day, x - 5, y + 10, 22);
355                     if (hasReminder(day + "" + (month + 1) + "" + year)) {
356                         drawRect(x + 18, y + 18, 24, 24, "red");
357                     }
358                 };
359                 drawPositionRect(icoord, jcoord, 30, 30, addRem, enlarge, days);
360                 writeSomething(days, icoord + 5, jcoord + 10, 8);
361                 //console.error(days + " " + month + " " + year);
362                 if (hasReminder(days + "" + (month + 1) + "" + year)) {
363                     //var remName = displayReminder(days, month, year);
364                     //writeSomething(remName, icoord+5, jcoord+25, 8);
365                     drawRect(icoord + 18, jcoord + 18, 12, 12, "red");
366                 }
367             }
368         }
369     }
370 }

372 /**
373  * Creates a reminder when a calendar day is double-clicked on.
374  * Currently working on being able to have user entered text and
375  * date/times.
376  *
377  * Alerts to be done later, along with recurring events.
378  * The input parameter is the Day that was clicked on. Used to
379  * automatically add a reminder to an associative
380  * array.
381  * 
382  * @param {int} day 
383  */
384 function addReminder(day) {
385     //ADDING A REMINDER!
386     var key = day + "" + (month + 1) + "" + year;
387     //Store current key in global variable so we know where to store reminder text
388     currentKey = key;
389     //Offset "Magic Numbers" - note the NewEvent call can't use these
390     //and is hardcoded
391     var backXOffset = 22;
392     var textXOffset = 25;
393 
394     if (!hasReminder(key)) {
395         reminders[key] = new Reminder(day, month + 1, year);
396         reminders[key].addName("reminder" + remNum++);
397 
398         var reminderDate = reminders[key].print();
399 
400         //Saving and restoring canvas context doesn't work
401         // as we've not actually drawn anything there. It's
402         // all javascript, so I just reinitialise the prototype
403         // instead. We'll need to create a custom script to "store"
404         // the current month they were on last to send them back to that.
405         var canvasreminder = 'canvas_1';
406 
407         $.get("emulatorBasics.js", function () {
408             newCanvas(320, 320, canvasreminder, true);
409             saveState(month, startDay, year);
410             var returnFunc = function () {
411                 returnToCalendar(savedMonth, savedDay, savedYear);
412             };
413             drawColourRect(20, cHeight - 30, 30, 15, returnFunc, true, "#FFFFFF");
414             writeSomethingColour("Back", backXOffset, cHeight - 20, 12, "black");
415             writeSomethingColour("New Event for " + reminderDate, textXOffset, 35, "20", "black");
416         });
417 
418     } else if (hasReminder(key)) {
419         //If there are reminders, show an Event View of the day clicked on.
420         var reminderDate = day + " " + monthToString(month) + " " + year;
421         var canvasreminder = 'canvas_1';
422 
423         $.get("emulatorBasics.js", function () {
424             newCanvas(320, 320, canvasreminder, false);
425             saveState(month, startDay, year);
426             //Return Button
427             var returnFunc = function () {
428                 returnToCalendar(savedMonth, savedDay, savedYear);
429             };
430             drawColourRect(20, cHeight - 30, 30, 15, returnFunc, true, "#FFFFFF");
431             writeSomethingColour("Back", backXOffset, cHeight - 20, 12, "black");
432 
433             //The "add new event" button
434             var newEvent = function () {
435                 newCanvas(320, 320, canvasreminder, true);
436 
437                 var returnFunc = function () {
438                     returnToCalendar(savedMonth, savedDay, savedYear);
439                 };
440                 drawColourRect(20, cHeight - 30, 30, 15, returnFunc, true, "#FFFFFF");
441                 writeSomethingColour("Back", 22, cHeight - 20, 12, "black");
442                 writeSomethingColour("New Event for " + reminderDate, 25, 35, "20", "black");
443             };
444 
445             drawColourRect(cWidth - 50, cHeight - 30, 30, 15, newEvent, true, "#FFFFFF");
446             writeSomethingColour("New", cWidth - 47, cHeight - 20, 12, "black");
447             //What day is it?
448             writeSomethingColour("Events for " + reminderDate, textXOffset, 35, "20", "black");
449 
450             //Show all events on that day.
451             var i;
452             for (i = 0; i < reminders[key].reminders.length; i++) {
453                 var savedReminders = reminders[key].reminders[i] + " - at: " + reminders[key].times[i];
454                 writeSomethingColour(savedReminders, textXOffset, 60 + (i * 25), "15", "white");
455             }
456         });
457     }
458     //REMINDER COMPLETE!
459 }

501 /**
502  * Find how many days in the month, given month and year.
503  * 
504  * @param {int} month
505  * @param {int} year
506  * @returns {int}
507  */
508 function daysInMonth(month, year) {
509     if (month === 1 && (year % 4 === 0) && ((year % 100 !== 0) || (year % 400 === 0))) {
510         return 29;
511     } else if (month === 1) {
512         return 28;
513     } else if (month === 3 || month === 5 || month === 8 || month === 10) {
514         //console.error("30 days this month" + month);
515         return 30;
516     } else {
517         //console.error("31 days this month" + month);
518         return 31;
519     }
520 }

522 /**
523  * This is the function used by clicking the
524  * Next Month button to advance the Calendar by
525  * a Month.
526  *
527  */
528 function advanceMonth() {
529     startDay = (startDay + daysInMonth(month, year)) % 7;
530     if (month < 11) {
531         month++;
532         changeMonth(month);
533     } else {
534         month = 0;
535         changeMonth(month);
536         year++;
537     }
538     var newDays = daysInMonth(month, year);
539     refreshInit(newDays, startDay);
540 }

542 /**
543  * Reverses the month by calculating a start day as an offset and populating
544  * the calendar with however many days are in that month.
545  *
546  */
547 function reverseMonth() {
548     //console.error("reversal: " + month + " " + startDay);
549     if (month > 0) {
550         month--;
551         changeMonth(month);
552     } else {
553         month = 11;
554         changeMonth(month);
555         year--;
556     }
557     newDays = daysInMonth(month, year);
558     tmp = startDay - newDays;
559     startDay = (((tmp % 7) + 7) % 7); //Sneaky negative modulo trick!
560     refreshInit(newDays, startDay);
561 }

596 /**
597  * Changes an integer month value into the name/string version
598  *
599  * @param {int} month
600  * @returns {String}
601  */
602 function monthToString(month) {
603     var stringMonths = ["January", "February", "March", "April", "May",
604         "June", "July", "August", "September", "October", "November",
605         "December"];
606     if (month < 12 && month >= 0) {
607         return stringMonths[month];
608     } else {
609         console.error("Error: Month number out of range (0-11)");
610         return "Invalid month number";
611     }
612 }

614 /**
615  * Changes string into integer value for any month
616  *
617  * @param {String} monthString
618  * @returns {int}
619  */
620 function monthToInt(monthString) {
621     var stringMonths = ["January", "February", "March", "April", "May",
622         "June", "July", "August", "September", "October", "November",
623         "December"];
624     for (i = 0; i < 12; i += 1) {
625         if (stringMonths[i] === monthString) {
626             return i;
627         }
628     }
