1 /** 
  2  * This class defines a reminder "class" to store a reminder for each day as necessary.
  3  * 
  4  * @module reminder.js
  5  * @author Ben Ryan
  6  */
  7 
  8 /**
  9  * 
 10  * @type @new; {Reminder}
 11  * 
 12  * @param {int} day
 13  * @param {int} month 
 14  * @param {int} year
 15  */
 16 function Reminder(day, month, year) {
 17     this.year = year;
 18     this.month = month;
 19     this.day = day;
 20     this.reminders = [];
 21     this.times = [];
 22     this.defaultTime = "0800";
 23     this.text = "";
 24     this.name = "";
 25 
 26     /**
 27      * Returns a representation of the reminder class as a string
 28      *  
 29      * @returns {String}
 30      */
 31     this.print = function () {
 32         var s = this.day + "-" + this.month + "-" + this.year;
 33         return s;
 34     };
 35 
 36     /**
 37      * Mutator to data field 'text'
 38      * 
 39      * @param {String} string
 40      */
 41     this.addText = function (string) {
 42         this.text = string;
 43     };
 44 
 45     /**
 46      * Mutator for name data field
 47      * @param {String} string
 48      */
 49     this.addName = function (string) {
 50         this.name = string;
 51     };
 52 
 53     /**
 54      * Accessor to Name data field
 55      * @returns {String}
 56      */
 57     this.returnName = function () {
 58         return this.name;
 59     };
 60 
 61     /**
 62      * Add a new reminder to reminders data field with default time to times data field
 63      * @param {String} reminder
 64      */
 65     this.newReminder = function (reminder) {
 66         if (this.reminders.length < 9) {
 67             this.reminders.push(reminder);
 68             this.times.push(this.defaultTime);
 69         } else if (this.reminders.length === 9) {
 70             this.reminders.push('Reminder limit for this date reached.');
 71         } else {
 72             console.error("No more reminders may be added for" + this.print());
 73         }
 74     };
 75 
 76     /**
 77      * Add a new time and new reminder to times and reminders respectively
 78      * @param {String} reminder
 79      * @param {String} time
 80      */
 81     this.newReminderWithTime = function (reminder, time) {
 82         if (this.reminders.length < 9) {
 83             this.reminders.push(reminder);
 84             this.times.push(time);
 85         } else if (this.reminders.length === 9) {
 86             this.reminders.push('Reminder limit for this date reached.');
 87         } else {
 88             console.error("No more reminders may be added for " + this.print());
 89         }
 90     };
 91     
 92     /**
 93      * Delete a reminder at the given index. Or complain if index is out of range.
 94      * @param {int} index
 95      */
 96     this.deleteReminder = function (index) {
 97         if(this.reminders.length < index) {
 98             reminders.splice(index, 1);
 99             times.splice(index, 1);
100         }else{
101             console.error("Reminder to be deleted does not exist! Index out of range.");
102         }
103     };
104 }
