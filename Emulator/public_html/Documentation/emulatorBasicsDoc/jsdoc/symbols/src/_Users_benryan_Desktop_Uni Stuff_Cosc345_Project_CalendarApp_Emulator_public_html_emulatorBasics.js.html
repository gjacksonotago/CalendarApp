 55 /**
 56  * Function get the mouse position
 57  * @param {Element} canvas 
 58  * @param {event} evt 
 59  * @return {object} object with x and y data fields 
 60  */
 61 function getMousePos(canvas, evt) {
 62     var rect = canvas.getBoundingClientRect();
 63     return {
 64         x: evt.clientX - rect.left,
 65         y: evt.clientY - rect.top
 66     };
 67 }

260 /**
261  * Adds a MouseClick event listener across the given
262  * co-ordinates - which it works out like the fillRectangle 
263  * method does, from a given (x, y) origin point and the size
264  * of the area to make interactive.
265  * 
266  * Utilises the canvas size properties to work the area out.
267  * 
268  * The event listener executes the function 'actionTaken'
269  * upon a mouseclick.
270  * 
271  * @param {int} xPosition
272  * @param {int} yPosition
273  * @param {int} xSize
274  * @param {int} ySize
275  * @param {function} actionTaken
276  */
277 function singleMouseClick(xPosition, yPosition, xSize, ySize, actionTaken) {
278     var canvas = document.getElementById('canvas_1');
279 
280     //The mouseclick event listener.
281     canvas.addEventListener('click', function (evt) {
282 
283         var mousePos = getMousePos(canvas, evt);
284 
285         var xCalc = (canvasWidth + xSize + xPosition) - canvasWidth;
286         var yCalc = (canvasHeight + ySize + yPosition) - canvasHeight;
287 
288         //If Mouse Clicked on the Black Square, new message!
289         if (mousePos.x <= xCalc && mousePos.y <= yCalc
290                 && mousePos.x >= (xPosition) && mousePos.y >= (yPosition)) {
291             actionTaken();
292         }
293     }, false);
294 }

399 /**
400  * Adds a swipe and call the function related to the event
401  * the swipe is define as a click on mouse held down then released at different position
402  * it could be left right up or down
403  * could be diagonal ?
404  * 
405  * @param {boolean} actionLeft
406  * @param {boolean} actionRight
407  * @param {boolean} actionUp
408  * @param {boolean} actionDown
409  * @returns {String}
410  */
411 function swipe(actionLeft, actionRight, actionUp, actionDown) {
412     var x1, x2, y1, y2;
413     var canvas = document.getElementById('canvas_1');
414     //The mousedown event listener.
415     canvas.addEventListener('mousedown', function (evt) {
416         var mousePos1 = getMousePos(canvas, evt);
417         x1 = mousePos1.x;
418         y1 = mousePos1.y;
419     });
420     //The mouseup event listener.
421     canvas.addEventListener('mouseup', function (evt) {
422         var mousePos2 = getMousePos(canvas, evt);
423         x2 = mousePos2.x;
424         y2 = mousePos2.y;
425         var dir = swipeDirection(x1, y1, x2, y2);
426         if (actionLeft !== false && dir === "left") {
427             actionLeft();
428         }
429         if (actionRight !== false && dir === "right") {
430             actionRight();
431         }
432         if (actionUp !== false && dir === "up") {
433             actionUp();
434         }
435         if (actionDown !== false && dir === "down") {
436             actionDown();
437         }
438         return dir;
439     });
440 }

444 /**
445  * Function returns the direction of swiping in the swiping area of the mouse
446  * 
447  * @param {boolean} actionLeft
448  * @param {boolean} actionRight
449  * @param {boolean} actionUp
450  * @param {boolean} actionDown
451  * @param {int} areaX
452  * @param {int} areaY
453  * @param {int} lenX
454  * @param {int} lenY 
455  * @returns {String}
456  */
457 function swipeArea(actionLeft, actionRight, actionUp, actionDown, areaX,
458         areaY, lenX, lenY) {
459     var x1, x2, y1, y2;
460     var canvas = document.getElementById('canvas_1');
461     //The mousedown event listener.
462 
463     canvas.addEventListener('mousedown', function (evt) {
464         var mousePos1 = getMousePos(canvas, evt);
465         x1 = mousePos1.x;
466         y1 = mousePos1.y;
467     });
468     if (x1 > areaX && x1 < areaX + lenX, y1 > areaY &&
469             y1 < areaY + lenY) {
470         //The mouseup event listener.
471         canvas.addEventListener('mouseup', function (evt) {
472             var mousePos2 = getMousePos(canvas, evt);
473             x2 = mousePos2.x;
474             y2 = mousePos2.y;
475 
476             var dir = swipeDirection(x1, y1, x2, y2);
477             if (actionLeft !== false && dir === "left") {
478                 actionLeft();
479             }
480             if (actionRight !== false && dir === "right") {
481                 actionRight();
482             }
483             if (actionUp !== false && dir === "up") {
484                 actionUp();
485             }
486             if (actionDown !== false && dir === "down") {
487                 actionDown();
488             }
489             return dir;
490         });
491     }
492 }

494 /*
495  * define the swipe direction could be 4 outputs only ( left right
496  * up and down ) no diagonal. the direction is defined between two 2D points  
497  * @param {int} x1
498  * @param {int} y1
499  * @param {int} x2
500  * @param {int} y2
501  * @returns {String}
502  */
503 function swipeDirection(x1, y1, x2, y2) {
504     var dir = "Invalid swipe";
505     var error = 50;//how far the mouse can sway in the other axis to main direction
506     if (x1 < x2 && Math.abs(y2 - y1) < error)
507         dir = "right";
508     else if (x2 < x1 && Math.abs(y2 - y1) < error)
509         dir = "left";
510     else if (y1 < y2 && Math.abs(x2 - x1) < error)
511         dir = "down";
512     else if (y2 < y1 && Math.abs(x2 - x1) < error)
513         dir = "up";
514     //console.log(dir); 
515     return dir;
516 }
